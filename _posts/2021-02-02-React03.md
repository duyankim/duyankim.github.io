# get

```javascript
const onClick = () => {
  axios.get("url").then((response) => {
    setData(response.data);
  });
};
```

# async

```javascript
  const onClick = async () => {
    try {
      const response = await axios.get(
        'url',
      );
      setData(response.data);
    } catch (e) {
      console.log(e);
    }
```
